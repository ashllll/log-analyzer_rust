// 在搜索结果回调中保存历史
_searchResultsSubscription = eventStreamService.searchResults.listen(
  (results) {
    // ... 现有逻辑 ...

    // 保存到搜索历史
    final workspaceId = ref.read(appStateProvider).activeWorkspaceId;
    if (workspaceId != null && _searchController.text.isNotEmpty) {

  },
