  // Optimistic update - immediately update UI
  state = AsyncData(previous.where((h) => h.query != query).toList());

  try {
    await bridge.deleteSearchHistory(query: query, workspaceId: workspaceId);

    // Rollback on failure

    rethrow;
  }
}
