```dart
  final value = _inputController.text.trim();
  if (value.isEmpty) return;

    value: value,
    operator_: widget.globalOperator,
    isRegex: false,
    priority: widget.terms.length,

  _inputController.clear();
}
```

```dart
String _buildPreview(List<SearchTermData> terms, QueryOperatorData op) {
  if (terms.isEmpty) return '无搜索条件';

  final opStr = op == QueryOperatorData.and ? ' AND ' :
                op == QueryOperatorData.or ? ' OR ' : ' NOT ';
  return terms.map((t) => t.value).join(opStr);
}
```
