```dart
/// 将 FFI VirtualTreeNodeData 转换为 Dart VirtualTreeNode
///
/// 递归处理嵌套的子节点
VirtualTreeNode _convertFromFfiNode(ffi.VirtualTreeNodeData node) {
  return switch (node) {
    ffi.VirtualTreeNodeData_File(:final name, :final path, :final hash, :final size, :final mimeType) =>
      VirtualTreeNode.file(
        name: name,
        path: path,
        hash: hash,
        size: size.toInt(),  // PlatformInt64 to int
        mimeType: mimeType,
      ),
    ffi.VirtualTreeNodeData_Archive(:final name, :final path, :final hash, :final archiveType, :final children) =>
      VirtualTreeNode.archive(
        name: name,
        path: path,
        hash: hash,
        archiveType: archiveType,
        children: children.map(_convertFromFfiNode).toList(),  // 递归转换
      ),
  };
}

/// 批量转换 FFI 节点列表
List<VirtualTreeNode> _convertFromFfiNodes(List<ffi.VirtualTreeNodeData> nodes) {
  return nodes.map(_convertFromFfiNode).toList();
}
```

Key points:
- Use Dart 3 pattern matching with switch expression
- Handle recursive children for archive nodes
- Convert PlatformInt64 to int for size field
