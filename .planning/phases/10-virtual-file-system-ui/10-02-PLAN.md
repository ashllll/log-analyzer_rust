Future<List<VirtualTreeNode>> loadChildren(String parentPath) async {
  // 调用后端加载子节点
  final children = await _getTreeChildrenViaBridge(
    bridge,
    workspaceId,
    parentPath,
  );
  // 更新树中对应节点的子节点
  final updatedTree = _updateNodeChildren(currentTree, parentPath, children);
  state = AsyncData(updatedTree);
  return children;
}

## TreeSliverController 使用模式 (来自 10-RESEARCH.md)
```dart
class _FileTreeViewState extends State<FileTreeView> {
  late TreeSliverController<String> _controller;

  @override
  void initState() {
    super.initState();
    _controller = TreeSliverController<String>();
  }

  Future<void> _handleExpand(VirtualTreeNode node) async {
    if (node.needsLazyLoad) {
      await widget.onLoadChildren(node.nodePath);
    }
    _controller.toggleExpansion(node.nodePath);
  }
}
```

  VirtualTreeNodeArchive(:final children) => children.isEmpty,
